	selection_tooltip = form_british_raj_decision_tooltip
	decision_group_type = major

	ai_check_interval = 120

	is_shown = {
		is_ruler = yes
		is_playable_character = yes

				culture = culture:gaelic
				culture = culture:english
				culture = culture:anglo_saxon
				culture = culture:welsh

			has_title = title:e_british_raj
		}
	}

		is_landed = yes
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
	}

	effect = {
		gain_heroic_legend_seed_tooltip_effect = yes
		custom_tooltip = form_british_raj_decision_effect_tooltip
		
		hidden_effect = {
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			
			title:e_british_raj = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
			resolve_title_and_vassal_change = scope:change
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}

				culture = culture:english
				culture = culture:anglo_saxon

				culture = culture:cornish

			title:e_rajastan = {

				}
			}
			create_title_and_vassal_change = {
				type = created
